  def self.sql_for_on_duplicate_key_update_as_array( arr )
    qt =  connection.quote_column_name( table_name )
    results = arr.map do |column|
      qc =  connection.quote_column_name( column )
      "#{qt}.#{qc}=VALUES( #{qc} )"        
    end
    results.join( ',' )
  end
